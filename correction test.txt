void ListeCapteurTemperature::suspendre(int id) { for (int i = 0; i < n; i++) {
if (tab[i].getId() == id) {
if (tab[i].getNiveauBatterie() < 30) { tab[i].setEtat("suspendu");

}

Evenement.cpp
#include "Evenement.h" #include <iostream> using namespace std;

// Definition de l'attribut statique (obligatoire dans le .cpp) int Evenement::nbTotalEvenements = 0;

Evenement::Evenement(int id, string titre, string date, int capacite)
: EvenementPreparation(id, titre, date)	// appel constructeur parent
{
capaciteMax = capacite;
nbTotalEvenements++;	// incrementation a chaque creation
}

Evenement::~Evenement() {
nbTotalEvenements--;	// decrementation a chaque destruction
}

void Evenement::afficher() { EvenementPreparation::afficher();	// affichage parent cout << "Capacite max : " << capaciteMax << endl;
cout << "Statut	: " << (capaciteMax == 0 ? "Complet" : "Disponible") << endl;
}

int Evenement::getCapaciteMax()	{ return capaciteMax; }
int Evenement::getNbTotal()	{ return nbTotalEvenements; }

#ifndef CENTRELOISIRS_H #define CENTRELOISIRS_H #include "Evenement.h"

CentreLoisirs.cpp
#include "CentreLoisirs.h" #include <iostream>
using namespace std;

CentreLoisirs::CentreLoisirs(string n) { nom = n;
this->n = 0;
}
 
// a. Ajouter avec verification unicite ID void CentreLoisirs::ajouter(Evenement e) {
for (int i = 0; i < n; i++) {
if (tab[i].getId() == e.getId()) {
cout << "Evenement avec ID " << e.getId() << " deja existant !" << endl; return;
}
}
tab[n] = e; n++;
cout << "Evenement ajoute." << endl;
}

// b. Afficher evenements par titre
void CentreLoisirs::afficherParTitre(string titre) {
cout << "=== Evenements avec le titre : " << titre << " ===" << endl; bool trouve = false;
for (int i = 0; i < n; i++) {
if (tab[i].getTitre() == titre) { tab[i].afficher();
cout << "---" << endl; trouve = true;
}
}
if (!trouve) cout << "Aucun evenement trouve." << endl;
}

// c. Supprimer les evenements complets (capaciteMax == 0) void CentreLoisirs::supprimerComplets() {
int i = 0; while (i < n) {
if (tab[i].getCapaciteMax() == 0) {
// decalage vers la gauche
for (int j = i; j < n-1; j++) { tab[j] = tab[j+1];
}
n--;
cout << "Evenement complet supprime." << endl;
} else {
i++;
}
}
}

// d. Afficher le nombre total (statique) void CentreLoisirs::afficherNbTotal() {
cout << "Nombre total d'evenements : " << Evenement::getNbTotal() << endl;
}
